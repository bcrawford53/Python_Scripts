#Device class creation to hold the attributes needed for SSH and commands
class NetworkDevice:
      def __init__(self, dev_type, devIP, usr="cisco", pwd="cisco",enable_pwd="cisco",command_list=[],host="x"):

            self.devIP = devIP

            self.pwd = pwd

            self.command_list= command_list
            self.host = host
